# –ü—Ä–æ–≤–µ—Ä–∏—Ç—å, –≤—Å–µ–≥–¥–∞ –ª–∏ h –≤—ã–±–∏—Ä–∞–µ—Ç—Å—è –∏–∑ –ø–æ–¥–≥—Ä—É–ø–ø—ã –≤ —Ç–µ–∫—É—â–∏—Ö —É—Ä–æ–≤–Ω—è—Ö

import json
from pathlib import Path

def check_layer4_h_validation():
    """
    –ü—Ä–æ–≤–µ—Ä—è–µ—Ç, –≤—Å–µ–≥–¥–∞ –ª–∏ witness.h –ø—Ä–∏–Ω–∞–¥–ª–µ–∂–∏—Ç –ø–æ–¥–≥—Ä—É–ø–ø–µ.
    –ï—Å–ª–∏ –Ω–µ—Ç ‚Äî —ç—Ç–æ –æ—à–∏–±–∫–∞ –≤ –¥–∞–Ω–Ω—ã—Ö!
    """
    levels_dir = Path("TheSymmetryVaults/data/levels/act1")
    issues = []

    for level_file in sorted(levels_dir.glob("level_*.json")):
        with open(level_file, 'r', encoding='utf-8') as f:
            data = json.load(f)

        level_num = int(level_file.stem.split('_')[1])
        layer4 = data.get("layers", {}).get("layer_4", {})
        subgroups = layer4.get("subgroups", [])

        for i, sg in enumerate(subgroups):
            elements = sg.get("elements", [])
            witness = sg.get("conjugation_witness")

            if witness is None:
                continue  # –ù–æ—Ä–º–∞–ª—å–Ω–∞—è –ø–æ–¥–≥—Ä—É–ø–ø–∞, witness –Ω–µ –Ω—É–∂–µ–Ω

            h = witness.get("h")
            if h not in elements:
                issues.append({
                    "level": level_num,
                    "subgroup_idx": i,
                    "elements": elements,
                    "witness_h": h,
                    "error": "h NOT in subgroup!"
                })

    if issues:
        print(f"‚ùå Found {len(issues)} witness issues:")
        for issue in issues:
            print(f"  Level {issue['level']}, subgroup {issue['subgroup_idx']}: "
                  f"h='{issue['witness_h']}' not in {issue['elements']}")
    else:
        print("‚úÖ All witnesses have h ‚àà H (data is correct)")

    return issues

if __name__ == "__main__":
    check_layer4_h_validation()
